 23 trait Scaling

 25     /**
 26      * fitInBounds
 27      *
 28      * @param int $w maximum width
 29      * @param int $h maximum height
 30      * @param int $width
 31      * @param int $height
 32      *
 33      * @access protected
 34      * @return mixed
 35      */
 36     protected function fitInBounds($w, $h, $width, $height)
 37     {
 38         if ($width >= $w or $height >= $h) {
 39 
 40             $ratA = $this->ratio($width, $height);
 41             $ratM = min($ratA, $this->ratio($w, $h));
 42 
 43             $isR = $ratM === $ratA;
 44 
 45             $valW = (int)round($h * $ratM);
 46             $valH = (int)round($w / $ratA);
 47 
 48             list($width, $height) = $width <= $height ? [$valW, $isR ? $h : $valH] : [$isR ? $w : $valW, $valH];
 49         }
 50 
 51         return compact('width', 'height');
 52     }

 54     /**
 55      * fillArea
 56      *
 57      * @param int $w
 58      * @param int $h
 59      * @param int $width
 60      * @param int $height
 61      *
 62      * @access protected
 63      * @return void
 64      */
 65     protected function fillArea(&$w, &$h, $width, $height)
 66     {
 67         extract($this->getInfo());
 68 
 69         $ratio = $this->ratio($width, $height);
 70 
 71         $minW = min($w, $width);
 72         $minH = min($h, $height);
 73         $minB = min($w, $h);
 74 
 75         if ($minB === 0 or ($minW > $width and $minH > $height)) {
 76             return;
 77         }
 78 
 79         $ratioC = $this->ratio($w, $h);
 80 
 81         list($w, $h) = $ratio > $ratioC ? [(int)ceil($h * $ratio), $h] : [$w, (int)ceil($w / $ratio)];
 82     }

138      * getCropCoordinates
139      *
140      * @param int $width   actual image width

142      * @param int $w       crop width
143      * @param int $h       crop height
144      * @param int $gravity crop position
145      *
146      * @access protected
147      * @return array an array containing the crop coardinates x and y
148      */
149     protected function getCropCoordinates($width, $height, $w, $h, $gravity)
150     {
151         $x = $y = 0;
152 
153         switch ($gravity) {
154         case 1: // GRAVITY_NORTHWEST
155             break;
156         case 3: // GRAVITY_NORTHEAST
157             $x = ($width) - $w;
158             break;
159         case 2: // GRAVITY_NORTH
160             $x = ($width / 2) - ($w / 2);
161             break;
162         case 4: // GRAVITY_WEST:
163             $y = ($height / 2) - ($h / 2);
164             break;
165         case 5: // GRAVITY_CENTER
166             $x = ($width / 2) - ($w / 2);
167             $y = $height / 2  - ($h / 2);
168             break;
169         case 6: // GRAVITY_EAST
170             $x = $width - $w;
171             $y = ($height / 2)  - ($h / 2);
172             break;
173         case 7: // GRAVITY_SOUTHWEST
174             $x = 0;
175             $y = $height - $h;
176             break;
177         case 8: // GRAVITY_SOUTH
178             $x = ($width / 2) - ($w / 2);
179             $y = $height - $h;
180             break;
181         case 9: // GRAVITY_SOUTHEAST
182             $x = $width - $w;
183             $y = $height - $h;
184             break;
185         }
186 
187         $x = (int)ceil($x);
188         $y = (int)ceil($y);
189 
190         return compact('x', 'y');
