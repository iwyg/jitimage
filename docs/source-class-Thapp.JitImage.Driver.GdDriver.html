181     private function loadResourceFromType($source)
182     {
183 
184         $type = getimagesize($source);
185 
186         $fn = sprintf('imagecreatefrom%s', $type = substr($type['mime'], strpos($type['mime'], '/') + 1));
187 
188         if (!function_exists($fn)) {
189

191             $this->clean();
192             return false;
193 
194         }
195 
196         $this->source = $this->loader->load($source);
197         $this->outputType = $type;
198 
199         return call_user_func($fn, $this->source);

244     /**
245      * resize
246      *
247      * @param mixed $width
248      * @param mixed $height
249      * @param mixed $flag
250      * @access protected
251      * @return mixed
252      */
253     protected function resize($width, $height, $flag = null)
254     {
255 
256         if (0 === min($width, $height)) {
257             extract($this->getTargetSize());
258         }
259 
260         switch ($flag) {
261         case static::FL_IGNR_ASPR:
262             break;
263         // oversize image to fill the boundaries.
264         case static::FL_FILL_AREA:
265             $this->fillArea($width, $height, $this->getInfo('width'), $this->getInfo('height'));
266             break;
267         // No scaling for larger images.
268         // Would be easier to just set `bestfit`, but its behaviour changed
269         // with imagemagick 3.0, so we have to calculate the best fit ou selfs.
270         case static::FL_OSRK_LGR:
271             extract($this->fitInBounds($width, $height, $this->getInfo('width'), $this->getInfo('height')));
272             break;
273         case static::FL_RESZ_PERC:
274             extract($this->percentualScale($this->getInfo('width'), $this->getInfo('height'), $width));
275             break;
276         case static::FL_PIXL_CLMT:
277             extract($this->pixelLimit($this->getInfo('width'), $this->getInfo('height'), $width));
278             break;
279         // default set the appropiate height and take width as a fixure in case
280         // both the image ratio and the resize ratio don't match
281         default:
282             $r1 = $this->getInfo('ratio');
283             $r2 = $this->ratio($width, $height);
284 
285             if (0.001 < abs($r1 - $r2)) {
286                 extract($this->getImageSize($width, 0));
287             }
288             break;
289         }
290 
291 
292         $resized = imagecreatetruecolor($width, $height);
293 
294         imagecopyresampled($resized, $this->resource, 0, 0, 0, 0, $width, $height, $this->getInfo('width'), $this->getInfo('height'));
295 
296         $this->swapResource($resized);
297         return $this;
298     }
