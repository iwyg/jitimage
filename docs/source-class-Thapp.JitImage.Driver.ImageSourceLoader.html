 53     /**
 54      * load
 55      *
 56      * @param string $url file source url
 57      *
 58      * @access public
 59      * @throws \Thapp\Exception\ImageResourceLoaderException
 60      * @return string
 61      */
 62     public function load($url)
 63     {
 64         if (file_exists($url)) {
 65             return $this->validate($url);
 66         }
 67 
 68         if (preg_match('#^(https?|spdy)://#', $url)) {
 69 
 70             if ($file = $this->loadRemoteFile($url)) {
 71                 return $this->validate($file);
 72             }
 73         }
 74 
 75         throw new ImageResourceLoaderException(sprintf('Invalid Source URL: %s', $url));
 76     }

 82      * @access private
 83      * @return mixed
 84      */
 85     private function validate($url)

 87         if (@getimagesize($url)) {
 88             return $url;
 89         }
 90 
 91         return false;
 92     }

117     /**
118      * loadRemoteFile
119      *
120      * @param mixed $url
121      * @access protected
122      * @return mixed
123      */
124     protected function loadRemoteFile($url)
125     {
126         $this->file = tempnam($this->tmp, 'jit_rmt_');
127 
128         if (!function_exists('curl_init')) {
129 
130             if (!$contents = file_get_contents($url)) {
131                 return false;
132             }
133 
134             file_put_contents($contents, $this->file);
135 
136             return $this->file;
137 
138         }
139 
140         $handle = fopen($this->file, 'w');
141 
142         if (!$this->fetchFile($handle, $url)) {
143             fclose($handle);
144             return false;
145         }
146 
147         fclose($handle);
148 
149         return $this->file;
150     }

152     /**
153      * fetchFile
154      *
155      * @param Resource $handle
156      * @access protected
157      * @return void
158      */
159     protected function fetchFile($handle, $url, &$message = null)
160     {
161 
162         $curl = curl_init($url);
163 
164         curl_setopt($curl, CURLOPT_RETURNTRANSFER, false);
165         curl_setopt($curl, CURLOPT_HEADER, false);
166         curl_setopt($curl, CURLOPT_FILE, $handle);
167 
168         $status = curl_exec($curl);
169         $info = curl_getinfo($curl);
170 
171         if (!in_array($info['http_code'], [200, 302])) {
172             $status = false;
173         }
174 
175         if (0 !== strlen($msg = curl_error($curl))) {
176 
177             $message = $msg;
178             $status = false;
179         }
180 
181         curl_close($curl);
182         return $status;
183     }
